import { Component, OnInit } from '@angular/core';
import { NgForm } from '@angular/forms';
import { NgxSpinnerService } from 'ngx-spinner';
import { ConvocatoriaPService } from 'src/app/services/convocatoria-p.service';
import { RespuestaServidor } from 'src/app/class/respuesta-servidor';
import swal from 'sweetalert2';
import { PlazasPService } from 'src/app/services/plazas-p.service';
import { Router } from '@angular/router';
import { Api } from 'src/app/class/api';
import { ConvocatoriaRefrigerioService } from 'src/app/services/convocatoria-refrigerio.service';

declare var jQuery:any;
declare var $:any;

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.css']
})
export class HomeComponent implements OnInit {

  user: any = {
    codigo: null
  };
  convocatoria:any = {};
  mostrarEntrevistas:Boolean = false;
  finInscripcion:Date;
  finEntrevista:Date;
  fechaTemporal:Date = new Date();
  etapaTemporal:any = null;
  api=Api.dev;
  urlDoc: string;
  convocatoriasRefrigerio:any[] = [];

  constructor(private spinner: NgxSpinnerService, private _convocatoriaP:ConvocatoriaPService, private convocatoriaRefrigerios : ConvocatoriaRefrigerioService, private _plazaP:PlazasPService, private router:Router) {

  }

  ngOnInit() {
    this.spinner.show();

    this._convocatoriaP.getConvocatoriaActiva().subscribe(data => {
      this.convocatoria = data['convocatoria'];
      // let today = Date.now();
      let today = new Date(data['fechaActual']+ "T00:00:00Z").getTime();
      if(this.convocatoria != null){
        if(this.convocatoria.etapaActual == null){

          this.convocatoria.etapas = this.convocatoria.etapas.filter(function(a) {
            return a.id > 3;
          });
    
            
          // let testDates = [{fechaFin:"2019-02-07"}, {fechaFin:"2019-02-13"}, {fechaFin:"2019-02-14"},{fechaFin:"2019-02-18"},{fechaFin:"2019-02-19"}, {fechaFin:"2019-06-07"}, {fechaFin:"2019-06-10"}, {fechaFin:"2019-06-17"}]

          this.convocatoria.etapas.sort(function(a, b) {
            return new Date(b.fechaFin + "T00:00:00Z").getTime() - new Date(a.fechaFin + "T00:00:00Z").getTime();
          });
          
          this.convocatoria.etapaActual = this.convocatoria.etapas.filter(function(a) {
            return (today - new Date(a.fechaFin + "T00:00:00Z").getTime()) > 0;
          })[0];
          // if(this.convocatoria.etapaActual == undefined){
          //   this.convocatoria.etapaActual = this.convocatoria.etapas.filter(function(a) {
          //     return a.id == 4;
          //   })[0];
          // }
        }

        // if(this.convocatoria.etapas != null){
        //   for(let i = 0; i<this.convocatoria.etapas.length; i++){
        //     if(this.convocatoria.etapas[i].peso == 3){
        //       this.finInscripcion = this.convocatoria.etapas[i].fechaFin;
        //     }
        //     if(this.convocatoria.etapas[i].peso == 4){
        //       this.finEntrevista = this.convocatoria.etapas[i].fechaFin;
        //       if(this.convocatoria.etapaActual == null){
        //         this.convocatoria.etapaActual = this.convocatoria.etapas[i];
        //       }
        //     }
            
        //   }
          
        // }
      }
      
      
      if(data['fechaActual'] <= this.finEntrevista && data['fechaActual'] > this.finInscripcion){
        this.mostrarEntrevistas = true;
      }
      this.spinner.hide();
    }, err => {
      console.log(err)
      this.spinner.hide();
      let respuesta: RespuestaServidor = new RespuestaServidor(err.status);
      swal({
        type: 'error',
        title: 'Error',
        text: respuesta.msg,
      });
    });

    this.convocatoriaRefrigerios.getConvocatoriasInscripcion().subscribe((data:any)=>{
      this.convocatoriasRefrigerio = data;
      this.spinner.hide();
    });
  }

  submitAction(forma: NgForm) {
    if (!forma.valid) {
      return;
    }

    this.router.navigate(['/plazasConvocatorias', this.convocatoria['id'], this.user.codigo ]);

  }

  viewDoc( doc:string ){
    this.urlDoc = doc;
    $('#modelId').modal('show');
  }

}
