import { Component, OnInit, AfterViewChecked, OnDestroy } from '@angular/core';
import { UserService } from 'src/app/services/user.service';
import { NgxSpinnerService } from 'ngx-spinner';
import { UsuarioCafeteriaService } from 'src/app/services/usuario-cafeteria.service';
import { RespuestaServidor } from 'src/app/class/respuesta-servidor';
import swal from 'sweetalert2';

@Component({
  selector: 'app-principal',
  templateUrl: './principal.component.html',
  styleUrls: ['./principal.component.css']
})
export class PrincipalComponent implements OnInit{
  
  userService:UserService;
  beneficios:any[]=[];
  
  constructor(private _userService:UserService, private spinner: NgxSpinnerService,private _usuarioService:UsuarioCafeteriaService) {
    this.userService = _userService;
    console.log(this.userService.tipoMatch(['Ayudantía']));
  }

  ngOnInit() {
    this.spinner.show();
    if (this.userService.roleMatch(['AdminCafeteria'])  || this.userService.roleMatch(['AsistenteCafeteria']) || this.userService.roleMatch(['AdminRefrigerios'])) {
      this._usuarioService.getEntrega().subscribe(data => {
        if (data['Succes']) {
          this.beneficios = data["beneficios"];   
        }  
        this.spinner.hide();
      }, err=> {
        let respuesta:RespuestaServidor = new RespuestaServidor(err.status);
        swal({
          type: 'error',
          title: 'Oops...',
          text: respuesta.msg,
        });
        this.spinner.hide(); 
      });
    }
    

  }
  


}
